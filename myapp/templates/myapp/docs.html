Post request or Add post


let init = {
  method : "post",
  headers: {
    'Content-type': 'application/json; charset=UTF-8',
  },
  body : JSON.stringify({
    title : "What is CSS",
    description : "CSS stands for Cascading StyleSheet. CSS is used to give the style to the web pages.",
    userId: 1
  })
}

fetch( `{{url}}api/posts`, init)
.then( (response) => {     
  if (!response.ok) {
    const message = `An error has occured: ${response.status}`;
    console.log(message);
    throw new Error(message);
  }
    return response.json();
})
.then(data => console.log(data));
Output 
{
  "postId":101,
  "title":"What is CSS",
  "description":"CSS stands for Cascading StyleSheet. CSS is used to give the style to the web pages.",
  "userId":1,
  "message":"Data Created.!!"
}
